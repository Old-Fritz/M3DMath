CMAKE_MINIMUM_REQUIRED(VERSION 3.8)

# define source dirs
SET(TESTS_VM_VECTORS_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/Vectors)
SET(TESTS_VM_OPERATIONS_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/Operations)
IF(NOT ANDROID)
	SET(TESTS_VM_SIMD_VECTORS_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/SIMDVectors)
ENDIF()

# find all sources
FILE(GLOB TESTS_VM_VECTORS_SRCS ${TESTS_VM_VECTORS_SRC_DIR}/*.cpp)
FILE(GLOB TESTS_VM_OPERATIONS_SRCS ${TESTS_VM_OPERATIONS_SRC_DIR}/*.cpp)
FILE(GLOB TESTS_VM_SIMD_VECTORS_SRCS ${TESTS_VM_SIMD_VECTORS_SRC_DIR}/*.cpp)

# add sources to target
TARGET_SOURCES(${MATH_TESTS_TARGET} PRIVATE ${TESTS_VM_VECTORS_SRCS} ${TESTS_VM_OPERATIONS_SRCS} ${TESTS_VM_SIMD_VECTORS_SRCS})
