CMAKE_MINIMUM_REQUIRED(VERSION 3.8)

# define source dirs
SET(VM_VECTORS_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/Vectors/x86_64_ARM)
SET(VM_OPERATIONS_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/Operations/x86_64_ARM)
IF(NOT ANDROID)
	SET(VM_SIMD_VECTORS_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/SIMDVectors/x86_64)
ENDIF()

# find all sources
FILE(GLOB VM_VECTORS_SRCS ${VM_VECTORS_SRC_DIR}/*.cpp)
FILE(GLOB VM_OPERATIONS_SRCS ${VM_OPERATIONS_SRC_DIR}/*.cpp)
FILE(GLOB VM_SIMD_VECTORS_SRCS ${VM_SIMD_VECTORS_SRC_DIR}/*.cpp)

# add sources to target
TARGET_SOURCES(${M3DMATH_LIB_TARGET} PRIVATE ${VM_VECTORS_SRCS} ${VM_OPERATIONS_SRCS} ${VM_SIMD_VECTORS_SRCS})
