CMAKE_MINIMUM_REQUIRED(VERSION 3.8)

# define source dirs
SET(MM_MATRIX_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR}/Matrix/x86_64_ARM)
IF(NOT ANDROID)
	SET(MM_MATRIX_SRC_DIR_86_64 ${CMAKE_CURRENT_SOURCE_DIR}/Matrix/x86_64)
ENDIF()

# find all sources
FILE(GLOB MM_MATRIX_SRCS ${MM_MATRIX_SRC_DIR}/*.cpp ${MM_MATRIX_SRC_DIR_86_64}/*.cpp)

# add sources to target
TARGET_SOURCES(${M3DMATH_LIB_TARGET} PRIVATE ${MM_MATRIX_SRCS})
